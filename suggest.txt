```javascript
// SMS Parser Implementation
class UPITransactionParser {
  parseTransactionSMS(smsContent) {
    const patterns = {
      amount: /(?:Rs\.?\s*|INR\s*|‚Çπ\s*)(\d+(?:,\d{3})*(?:\.\d{2})?)/,
      merchant: /(?:to|paid)\s+([A-Za-z0-9\s&.-]+?)(?:\s+via|\s+using|\s+on)/i,
      upiId: /(\w+@\w+)/,
      timestamp: /(\d{2}[-/]\d{2}[-/]\d{4}\s+\d{2}:\d{2})/
    };
    
    return {
      amount: this.extractAmount(smsContent),
      merchant: this.extractMerchant(smsContent),
      roundUpAmount: this.calculateRoundUp(amount),
      timestamp: new Date()
    };
  }
  
  calculateRoundUp(amount) {
    const nextTen = Math.ceil(amount / 10) * 10;
    return nextTen - amount;
  }
}
```
